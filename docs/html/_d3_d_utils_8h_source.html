187 static UINT CalcConstantBufferByteSize(UINT byteSize)
188 {
189 // Constant buffers must be a multiple of the minimum hardware
190 // allocation size (usually 256 bytes). So round up to nearest
191 // multiple of 256. We do this by adding 255 and then masking off
192 // the lower 2 bytes which store all bits < 256.
193 // Example: Suppose byteSize = 300.
194 // (300 + 255) & ~255
195 // 555 & ~255
196 // 0x022B & ~0x00ff
197 // 0x022B & 0xff00
198 // 0x0200
199 // 512
200 return (byteSize + 255) & ~255;
201 }
