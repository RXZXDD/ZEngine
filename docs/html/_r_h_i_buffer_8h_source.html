16 static uint32 CalcConstantBufferByteSize(uint32 byteSize)
17 {
18 // Constant buffers must be a multiple of the minimum hardware
19 // allocation size (usually 256 bytes). So round up to nearest
20 // multiple of 256. We do this by adding 255 and then masking off
21 // the lower 2 bytes which store all bits < 256.
22 // Example: Suppose byteSize = 300.
23 // (300 + 255) & ~255
24 // 555 & ~255
25 // 0x022B & ~0x00ff
26 // 0x022B & 0xff00
27 // 0x0200
28 // 512
29 return (byteSize + 255) & ~255;
30 }
